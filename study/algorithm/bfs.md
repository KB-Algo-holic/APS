from collections import deque

def bfs(graph, start):
    visited = set()
    queue = deque([start])
    visited.add(start)
    
    while queue:
        v = queue.popleft()
        print(v, end=' ')
        for next_v in graph[v]:
            if next_v not in visited:
                visited.add(next_v)
                queue.append(next_v)
